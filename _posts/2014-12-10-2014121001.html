public class PoissonEvaluator { decimal lambda;//均值 public PoissonEvaluator(decimal lambda = 1.0M) { this.lambda = lambda; } public decimal ProbabilityMassFunction(int k)//求出现k的概率 { //(l^k / k! ) * e^-l //l = lamda int kFactorial = Factorial(k); double numerator = Math.Pow(Math.E, -(double)lambda) * Math.Pow((double)lambda, (double)k); decimal p = (decimal)numerator / kFactorial; return p; } public decimal CummulitiveDistributionFunction(int k)//求出现不大于k的概率 { double e = Math.Pow(Math.E, (double)-lambda); int i = 0; double sum = 0.0; while (i <= k) { double n = Math.Pow((double)lambda, i) / Factorial(i); sum += n; i++; } decimal cdf = (decimal)e * (decimal)sum; return cdf; } private int Factorial(int k) { int count = k; int factorial = 1; while (count >= 1) { factorial = factorial * count; count--; } return factorial; } }

C# 批量处理SQLite中的数据

public int InsertDataFromDataTable(System.Data.DataTable dt) { int result = 0; SQLiteConnection conn = new SQLiteConnection(this.dbPath); conn.Open(); SQLiteCommand cmd = new SQLiteCommand(conn);//实例化SQL命令 SQLiteTransaction tran = conn.BeginTransaction(); cmd.Transaction = tran; try { for (int i = 0; i < dt.Rows.Count; i++) { System.Data.DataRow dr = dt.Rows[i]; cmd.Parameters.Clear(); cmd.CommandText = @"Insert into table1(col1,col2) select @col1, @col2";//设置带参SQL语句 ; cmd.Parameters.AddRange( new[] {//添加参数 new SQLiteParameter("@col1", dr["col1"].ToString()), new SQLiteParameter("@col2", dr["col2"].ToString()) }); result += cmd.ExecuteNonQuery();//执行 } tran.Commit();//提交 } catch (Exception) { tran.Rollback(); } conn.Close(); return result; }
